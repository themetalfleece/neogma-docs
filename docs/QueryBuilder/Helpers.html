linkHelpers

The QueryBuilder class also provides some helpers for generating strings which could be used in a statement.

linkGetting normalized labels

QueryRunner.getNormalizedLabels returns a single string to be used in a query.

1const { getNormalizedLabels } = QueryRunner;

2

3console.log(getNormalizedLabels('Users')); // `Users`

4

5console.log(getNormalizedLabels('Important Users')); // "`Important Users`"

6

7console.log(getNormalizedLabels(['Users', 'Active'])); // "`Users:Active`"

8

9console.log(getNormalizedLabels(['Users', 'Active'], 'or')); // "`Users|Active`"

10

11console.log(getNormalizedLabels(['Users', 'Active', 'Old'])); // "`Users:Active:Old`"

linkGetting a node statement

QueryRunner.getNodeStatement returns a string for a node's identifier, label, and inner info (like a where), to be used in a query.

Every parameter is optional.

1const { getNodeStatement } = QueryRunner;

2

3console.log(getNodeStatement({

4    identifier: 'n',

5    label: 'MyLabel'

6})); // (n:MyLabel)

7

8console.log(getNodeStatement({

9    identifier: 'n',

10})); // (n)

11

12console.log(getNodeStatement({

13    label: 'MyLabel',

14})); // (:MyLabel)

15

16console.log(getNodeStatement({})); // ()

17

18// --> an inner statement can be given

19console.log(getNodeStatement({

20    identifier: 'n',

21    label: 'MyLabel'

22    inner: '{ id: 1 }' // --> using a literal string as inner

23})); // (n:MyLabel { id: 1 })

24

25/* --> using a Where instance as inner */

26const where = new Where({ id: 1 });

27console.log(getNodeStatement({

28    identifier: 'n',

29    label: 'MyLabel'

30    inner: where 

31})); // (n:MyLabel { id: $id })

32console.log(where.getBindParam().get()); // { id: 1 }

33

34/* --> using a BindParam and a properties object instance as inner */

35const bindParam = new BindParam();

36console.log(getNodeStatement({

37    identifier: 'n',

38    label: 'MyLabel'

39    inner: {

40        properties: {

41            id: 1,

42        },

43        bindParam: bindParam

44    }

45})); // (n:MyLabel { id: $id })

46console.log(bindParam().get()); // { id: 1 }

linkGetting a relationship statement

QueryRunner.getRelationshipStatement returns a string for a relationship's direction, name, and inner info (like a where), to be used in a query.

1const { getRelationshipStatement } = QueryRunner;

2

3console.log(getRelationshipStatement({

4    direction: 'out',

5    name: 'HAS',

6    identifier: 'r'

7})); // -[r:HAS]->

8

9console.log(getRelationshipStatement({

10    direction: 'in',

11    name: 'HAS',

12    identifier: 'r'

13})); // <-[r:HAS]-

14

15console.log(getRelationshipStatement({

16    direction: 'none',

17    name: 'HAS',

18    identifier: 'r'

19})); // -[r:HAS]-

20

21console.log(getRelationshipStatement({

22    direction: 'out',

23    name: 'HAS',

24    // --> in any of the above cases, the identifier can be skipped

25})); // -[:HAS]->

26

27// --> an inner statement can be given

28console.log(getRelationshipStatement({

29    direction: 'out',

30    name: 'HAS',

31    identifier: 'r',

32    inner: '{ id: 1 }' // --> using a literal string as inner

33})); // -[r:HAS { id: 1}]->

34

35/* --> using a Where instance as inner */

36const where = new Where({ id: 1 });

37console.log(getRelationshipStatement({

38    direction: 'out',

39    name: 'HAS',

40    identifier: 'r',

41    inner: where 

42})); // -[r:HAS { id: $id }]->

43console.log(where.getBindParam().get()); // { id: 1 }

44

45/* --> using a BindParam and a properties object instance as inner */

46const bindParam = new BindParam();

47console.log(getRelationshipStatement({

48    direction: 'out',

49    name: 'HAS',

50    identifier: 'r',

51    inner: {

52        properties: {

53            id: 1,

54        },

55        bindParam: bindParam

56    }

57})); // -[r:HAS { id: $id }]->

58console.log(bindParam.get()); // { id: 1 }

linkGetting an identifier with a label

QueryRunner.getIdentifierWithLabel returns a string to be used in a query, regardless if any of the identifier or label are null

1const { getIdentifierWithLabel } = QueryRunner;

2

3console.log(getIdentifierWithLabel('MyIdentifier', 'MyLabel')); // "MyIdentifier:MyLabel"

4

5console.log(getIdentifierWithLabel('MyIdentifier', 'MyLabel')); // "MyIdentifier"

6

7console.log(getIdentifierWithLabel('MyIdentifier', 'MyLabel')); // ":MyLabel"

linkGetting parts for a SET operation

QueryRunner.getSetParts returns the parts and the statement for a SET operation.

1const { getSetParts } = QueryRunner;

2

3const existingBindParam = new BindParam({});

4const result = getSetParts({

5    /* --> the data to set */

6    data: {

7        x: 5,

8        y: 'foo'

9    },

10    /* --> BindParam instance to be used */

11    bindParam: existingBindParam, // @see [Bind Paramters](../Bind-Parameters)

12    /* --> the identifier to use */

13    identifier: 'node'

14});

15console.log(result.parts); // ["node.x = $x", "node.y = $y"]

16console.log(result.statement); // "SET node.x = $x, node.y = $y"

17console.log(bindParam.get()); // { x: 5, y: 'foo' }

18

19const existingBindParam = new BindParam({

20    x: 'irrelevant'

21});

22const result = getSetParts({

23    data: {

24        x: 5,

25        y: 'foo'

26    },

27    bindParam: existingBindParam,

28    identifier: 'node'

29});

30console.log(result.parts); // ["node.x = $x__aaaa", "node.y = $y"]

31console.log(result.statement); // "SET node.x = $x__aaaa, node.y = $y"

32console.log(bindParam.get()); // { x: 'irrelevant', x_aaaa: 5, y: 'foo' }

linkGetting properties with query param values

QueryRunner exposes a getPropertiesWithParams function which returns an object in a string format to be used in queries, while replacing its values with bind params.

1/* --> an existing BindParam instance, could have existing values */

2const bindParam = new BindParam({

3    x: 4,

4});

5const result = QueryRunner.getPropertiesWithParams(

6    /* --> the object to use */

7    {

8        x: 5,

9        y: 'foo'

10    },

11    /* --> an existing bindParam must be passed */

12    bindParam

13);

14

15/* --> the result gives us the needed object, while replacing its values with the appropriate bind param */

16console.log(result); // "{ x: $x__aaaa, y: $y }"

17console.log(bindParam.get()); // { x: 4, x__aaaa: 5, y: 'foo' }
